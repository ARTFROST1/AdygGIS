// Info Card
Surface(
    modifier = Modifier
        .fillMaxWidth()
        .shadow(1.dp, RoundedCornerShape(16.dp)),
    shape = RoundedCornerShape(16.dp),
    color = MaterialTheme.colorScheme.surface,
    border = BorderStroke(1.dp, outlineVariant)

    Row(padding = 12.dp) {
        Box(44.dp circle, 12.dp radius)  // Icon

        }
        Icon(Chevron, 18.dp)  // If clickable
    }

Surface(
    modifier = Modifier
        .fillMaxWidth()
        .clickable()
        .shadow(1.dp, RoundedCornerShape(14.dp)),

    color = primaryContainer,
    border = BorderStroke(1.5.dp, primary.copy(0.3f))
) {
    Row(padding = 12.dp) {
        Box(40.dp circle, 10.dp radius)  // Icon
        Column {
            Text(12.sp, semibold)   // Label
            Text(14.sp, medium)     // Value
        }
        Icon(Arrow, 20.dp)
    }
}

// Amenity Card
Surface(
    modifier = Modifier
        .weight(1f)
        .shadow(1.dp, RoundedCornerShape(12.dp)),
    shape = RoundedCornerShape(12.dp),
    color = surface,
    border = BorderStroke(1.dp, outlineVariant)
) {
    Row(padding = 12.dp/10.dp) {
        Box(32.dp circle)  // Icon
        Text(13.sp, medium, maxLines = 2)
    }
}

        .padding(14.dp/8.dp)

    Text(

    )
}
